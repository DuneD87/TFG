cmake_minimum_required(VERSION 3.3)
project(opengltest)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")
set(SOURCE_FILES main.cpp)
if (assimp_FOUND)
    set(ASSIMP_LIBRARY "assimp")
    add_library(${ASSIMP_LIBRARY} SHARED IMPORTED)
    set_target_properties(${ASSIMP_LIBRARY} PROPERTIES IMPORTED_LOCATION "${ASSIMP_LIBRARY_DIRS}/libassimp.so")
endif(assimp_FOUND)
add_executable(opengltest main.cpp glad.c GameClasses/Shader.cpp GameClasses/Shader.h stb_image.h GameClasses/Camera.cpp GameClasses/Camera.h GameClasses/Mesh.cpp GameClasses/Mesh.h GameClasses/Model.cpp GameClasses/Model.h GameClasses/Light.cpp GameClasses/Light.h GameClasses/XmlParser.cpp GameClasses/XmlParser.h GameClasses/BasicShapeBuilder.h GameClasses/Scene.cpp GameClasses/Scene.h)
target_link_libraries(opengltest glfw GL X11 pthread Xrandr Xi dl assimp)
